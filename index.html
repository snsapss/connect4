class Connect4AI { constructor(rows, cols) { this.R = rows; this.C = cols; this.board = Array.from({ length: rows }, () => Array(cols).fill(0)); this.TT = new Map(); // transposition table } moves() { let m = []; for (let c = 0; c < this.C; c++) { if (this.board[0][c] === 0) m.push(c); } return m; } drop(c, p) { for (let r = this.R - 1; r >= 0; r--) { if (this.board[r][c] === 0) { this.board[r][c] = p; return r; } } return -1; } undo(c) { for (let r = 0; r < this.R; r++) { if (this.board[r][c] !== 0) { this.board[r][c] = 0; break; } } } win(p) { let b = this.board, R = this.R, C = this.C; for (let r = 0; r < R; r++) { for (let c = 0; c < C; c++) { if (b[r][c] !== p) continue; if (c + 3 < C && b[r][c + 1] === p && b[r][c + 2] === p && b[r][c + 3] === p) return true; if (r + 3 < R && b[r + 1][c] === p && b[r + 2][c] === p && b[r + 3][c] === p) return true; if (r + 3 < R && c + 3 < C && b[r + 1][c + 1] === p && b[r + 2][c + 2] === p && b[r + 3][c + 3] === p) return true; if (r + 3 < R && c - 3 >= 0 && b[r + 1][c - 1] === p && b[r + 2][c - 2] === p && b[r + 3][c - 3] === p) return true; } } return false; } score() { let score = 0; let b = this.board, R = this.R, C = this.C; let center = Math.floor(C / 2); // center control (VERY important) for (let r = 0; r < R; r++) { if (b[r][center] === 2) score += 6; if (b[r][center] === 1) score -= 6; } // window scoring const evalWindow = (arr) => { let ai = arr.filter(x => x === 2).length; let hu = arr.filter(x => x === 1).length; if (ai === 4) return 1000; if (ai === 3 && hu === 0) return 50; if (ai === 2 && hu === 0) return 10; if (hu === 3 && ai === 0) return -80; return 0; }; for (let r = 0; r < R; r++) { for (let c = 0; c < C - 3; c++) { score += evalWindow([b[r][c], b[r][c + 1], b[r][c + 2], b[r][c + 3]]); } } for (let c = 0; c < C; c++) { for (let r = 0; r < R - 3; r++) { score += evalWindow([b[r][c], b[r + 1][c], b[r + 2][c], b[r + 3][c]]); } } return score; } key() { return this.board.map(r => r.join("")).join("|"); } orderMoves(moves) { let center = Math.floor(this.C / 2); return moves.sort((a, b) => Math.abs(a - center) - Math.abs(b - center) ); } minimax(depth, alpha, beta, maximizing) { let key = this.key() + depth + maximizing; if (this.TT.has(key)) return this.TT.get(key); if (this.win(2)) return 100000; if (this.win(1)) return -100000; if (depth === 0) return this.score(); let moves = this.orderMoves(this.moves()); if (maximizing) { let best = -Infinity; for (let c of moves) { this.drop(c, 2); let val = this.minimax(depth - 1, alpha, beta, false); this.undo(c); best = Math.max(best, val); alpha = Math.max(alpha, best); if (beta <= alpha) break; } this.TT.set(key, best); return best; } else { let best = Infinity; for (let c of moves) { this.drop(c, 1); let val = this.minimax(depth - 1, alpha, beta, true); this.undo(c); best = Math.min(best, val); beta = Math.min(beta, best); if (beta <= alpha) break; } this.TT.set(key, best); return best; } } bestMove() { let moves = this.orderMoves(this.moves()); let bestScore = -Infinity; let bestMove = moves[0]; for (let c of moves) { this.drop(c, 2); let score = this.minimax(6, -Infinity, Infinity, false); // 🔥 strong depth this.undo(c); if (score > bestScore) { bestScore = score; bestMove = c; } } return bestMove; } }
